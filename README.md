SELECT 
    /* 🔹 ON DATE */
    NVL(SUM(CASE 
        WHEN DESTINATION = 'LD1'
        AND DATE_TIME >= TO_DATE(:pDate,'DD/MM/YYYY') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS LD1_TONS,

    NVL(SUM(CASE 
        WHEN DESTINATION = 'LD2'
        AND DATE_TIME >= TO_DATE(:pDate,'DD/MM/YYYY') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS LD2_TONS,

    NVL(SUM(CASE 
        WHEN DESTINATION = 'LD3'
        AND DATE_TIME >= TO_DATE(:pDate,'DD/MM/YYYY') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS LD3_TONS,

    NVL(SUM(CASE 
        WHEN DESTINATION = 'MRD' AND TRP_NO <= 50
        AND DATE_TIME >= TO_DATE(:pDate,'DD/MM/YYYY') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS MRDTP_TONS,

    NVL(SUM(CASE 
        WHEN TRP_NO <= 50 AND RET_FLAG = 'N'
        AND DATE_TIME >= TO_DATE(:pDate,'DD/MM/YYYY') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN FILL_STATUS END), 0) AS NOOFTP,

    /* 🔹 MONTH CUMULATIVE */
    NVL(SUM(CASE 
        WHEN DESTINATION = 'LD1'
        AND DATE_TIME >= TRUNC(TO_DATE(:pDate,'DD/MM/YYYY'),'MON') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS LD1_TONS_ACTUAL,

    NVL(SUM(CASE 
        WHEN DESTINATION = 'LD2'
        AND DATE_TIME >= TRUNC(TO_DATE(:pDate,'DD/MM/YYYY'),'MON') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS LD2_TONS_ACTUAL,

    NVL(SUM(CASE 
        WHEN DESTINATION = 'LD3'
        AND DATE_TIME >= TRUNC(TO_DATE(:pDate,'DD/MM/YYYY'),'MON') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS LD3_TONS_ACTUAL,

    NVL(SUM(CASE 
        WHEN DESTINATION = 'MRD' AND TRP_NO <= 50
        AND DATE_TIME >= TRUNC(TO_DATE(:pDate,'DD/MM/YYYY'),'MON') + 6/24
        AND DATE_TIME <  TO_DATE(:pDate,'DD/MM/YYYY') + 1 + 6/24
    THEN NET_WT END), 0) AS MRDTP_TONS_ACTUAL

FROM demo.t_ladle
WHERE FUR_NAME = 'G';
