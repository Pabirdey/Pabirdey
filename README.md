SELECT * FROM (
    SELECT 
        Timestamp,
        SHIFT,
        RETURN_FINES_AL2O3,
        WET_FINES_AL2O3,
        DRY_FINES_AL2O3,
        DRY_FINES_500TPH_AL2O3 ,
        RETURN_FINES_SIO2,
        WET_FINES_SIO2,
        DRY_FINES_SIO2,
        DRY_FINES_500TPH_SIO2,
        RETURN_FINES_P,
        WET_FINES_P,
        DRY_FINES_P,
        DRY_FINES_500TPH_P,
        RETURN_FINES_K2O,
        WET_FINES_K2O,
        DRY_FINES_K2O,
        DRY_FINES_500TPH_K20
    FROM imtg.T_NOA_PILE_MAT_ANAL_SHIFT 
    WHERE SHIFT = (SELECT MAX(SHIFT) FROM imtg.T_NOA_PILE_MAT_ANAL_SHIFT)
    ORDER BY Timestamp DESC
) 
WHERE ROWNUM = 1;
