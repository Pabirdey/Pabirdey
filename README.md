        function CalcUlAndBalTotal() {
            debugger;
            var ulTotal = 0;
            var balanceTotal = 0;          
            $(".ul").each(function () {
                var val = parseFloat($(this).val()) || 0;
                ulTotal += val;
            });          
            $(".bal").each(function () {
                var val = parseFloat($(this).val()) || 0;
                balanceTotal += val;
            });    
            
            $("txtTotul").val();
            $("txtTotBal").val();
        }
