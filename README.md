SELECT 'A-F' AS PLANT, SUM(LD1_TONS_ACTUAL) LD1_TONS_ACT
FROM DEMO.T_LADLE
WHERE PLANT = 'A-F'
AND DATE_TIME >= TO_DATE('01-APR-2026','DD-MON-YYYY')
AND DATE_TIME <  TO_DATE('14-APR-2026','DD-MON-YYYY')

UNION ALL

SELECT 'A-G', SUM(LD1_TONS_ACTUAL)
FROM DEMO.T_LADLE
WHERE PLANT IN ('A-F','G')
AND DATE_TIME >= TO_DATE('01-APR-2026','DD-MON-YYYY')
AND DATE_TIME <  TO_DATE('14-APR-2026','DD-MON-YYYY')

UNION ALL

SELECT 'A-H', SUM(LD1_TONS_ACTUAL)
FROM DEMO.T_LADLE
WHERE PLANT IN ('A-F','G','H')
AND DATE_TIME >= TO_DATE('01-APR-2026','DD-MON-YYYY')
AND DATE_TIME <  TO_DATE('14-APR-2026','DD-MON-YYYY')

UNION ALL

SELECT 'A-I', SUM(LD1_TONS_ACTUAL)
FROM DEMO.T_LADLE
WHERE PLANT IN ('A-F','G','H','I')
AND DATE_TIME >= TO_DATE('01-APR-2026','DD-MON-YYYY')
AND DATE_TIME <  TO_DATE('14-APR-2026','DD-MON-YYYY')

ORDER BY PLANT;
